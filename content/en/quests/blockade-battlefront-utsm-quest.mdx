  <Step num="1" title="Accept the Special Titan Request">

  <Step num="2" title="Complete the Soundwave Survival Trial">

  </Step>
  <Step num="3" title="Farm Zombie Core Materials">

  </Step>
  <Step num="4" title="Reach Mastery Level 25">

  </Step>
  <Step num="5" title="Final Assembly and Activation">
    Return to the quest menu with 50,000 Cens and your collected materials. Confirm the assembly to permanently unlock UTSM in your Special Titans inventory.
  </Step>
</Steps>
